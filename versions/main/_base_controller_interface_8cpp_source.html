25rclcpp_lifecycle::node_interfaces::LifecycleNodeInterface::CallbackReturn BaseControllerInterface::on_init() {
26 // registering set_parameter callbacks is only possible on_init since the lifecycle node is not yet initialized
27 // on construction. This means we might not be able to validate parameter overrides - if they are provided.
28 pre_set_parameter_cb_handle_ = get_node()->add_pre_set_parameters_callback(
29 [this](std::vector<rclcpp::Parameter>& parameters) { return this->pre_set_parameters_callback(parameters); });
30 on_set_parameter_cb_handle_ = get_node()->add_on_set_parameters_callback(
31 [this](const std::vector<rclcpp::Parameter>& parameters) -> rcl_interfaces::msg::SetParametersResult {
32 return this->on_set_parameters_callback(parameters);
33 });
34
35 return CallbackReturn::SUCCESS;
36}

59void BaseControllerInterface::add_parameter(
60 const std::shared_ptr<ParameterInterface>& parameter, const std::string& description, bool read_only) {
61 rclcpp::Parameter ros_param;
62 try {
63 ros_param = translators::write_parameter(parameter);
64 } catch (const modulo_core::exceptions::ParameterTranslationException& ex) {
65 throw modulo_core::exceptions::ParameterException("Failed to add parameter: " + std::string(ex.what()));
66 }
67 if (!get_node()->has_parameter(parameter->get_name())) {
68 RCLCPP_DEBUG(get_node()->get_logger(), "Adding parameter '%s'.", parameter->get_name().c_str());
69 parameter_map_.set_parameter(parameter);
70 read_only_parameters_.insert_or_assign(parameter->get_name(), false);
71 try {
72 rcl_interfaces::msg::ParameterDescriptor descriptor;
73 descriptor.description = description;
74 descriptor.read_only = read_only;
75 // since the pre_set_parameters_callback is not called on parameter declaration, this has to be true
76 set_parameters_result_.successful = true;
77 set_parameters_result_.reason = "";
78 if (parameter->is_empty()) {
79 descriptor.dynamic_typing = true;
80 descriptor.type = translators::get_ros_parameter_type(parameter->get_parameter_type());
81 get_node()->declare_parameter(parameter->get_name(), rclcpp::ParameterValue{}, descriptor);
82 } else {
83 get_node()->declare_parameter(parameter->get_name(), ros_param.get_parameter_value(), descriptor);
84 }
85 std::vector<rclcpp::Parameter> ros_parameters{get_node()->get_parameters({parameter->get_name()})};
86 pre_set_parameters_callback(ros_parameters);
87 auto result = on_set_parameters_callback(ros_parameters);
88 if (!result.successful) {
89 get_node()->undeclare_parameter(parameter->get_name());
90 throw modulo_core::exceptions::ParameterException(result.reason);
91 }
92 read_only_parameters_.at(parameter->get_name()) = read_only;
93 } catch (const std::exception& ex) {
94 parameter_map_.remove_parameter(parameter->get_name());
95 read_only_parameters_.erase(parameter->get_name());
96 throw modulo_core::exceptions::ParameterException("Failed to add parameter: " + std::string(ex.what()));
97 }
98 } else {
99 RCLCPP_DEBUG(get_node()->get_logger(), "Parameter '%s' already exists.", parameter->get_name().c_str());
100 }

111void BaseControllerInterface::pre_set_parameters_callback(std::vector<rclcpp::Parameter>& parameters) {
112 if (pre_set_parameter_callback_called_) {
113 pre_set_parameter_callback_called_ = false;
114 return;
115 }
116 rcl_interfaces::msg::SetParametersResult result;
117 result.successful = true;
118 for (auto& ros_parameter : parameters) {
119 try {
120 auto parameter = parameter_map_.get_parameter(ros_parameter.get_name());
121 if (read_only_parameters_.at(ros_parameter.get_name())) {
122 RCLCPP_DEBUG(get_node()->get_logger(), "Parameter '%s' is read only.", ros_parameter.get_name().c_str());
123 continue;
124 }
125
126 // convert the ROS parameter into a ParameterInterface without modifying the original
127 auto new_parameter = translators::read_parameter_const(ros_parameter, parameter);
128 if (!this->validate_parameter(new_parameter)) {
129 result.successful = false;
130 result.reason += "Validation of parameter '" + ros_parameter.get_name() + "' returned false!";
131 } else if (!new_parameter->is_empty()) {
132 // update the value of the parameter in the map
133 translators::copy_parameter_value(new_parameter, parameter);
134 ros_parameter = translators::write_parameter(new_parameter);
135 }
136 } catch (const std::exception& ex) {
137 result.successful = false;
138 result.reason += ex.what();
139 }
140 }
141 set_parameters_result_ = result;
142}

146 auto result = set_parameters_result_;
147 set_parameters_result_.successful = true;
148 set_parameters_result_.reason = "";
149 return result;

183 RCLCPP_DEBUG(get_node()->get_logger(), "Adding predicate '%s'.", predicate_name.c_str());

186 this->predicates_.insert_or_assign(predicate_name, Predicate(predicate_function));
187 } catch (const std::exception& ex) {

189 get_node()->get_logger(), "Failed to evaluate callback of predicate '%s', returning false: %s",
190 predicate_name.c_str(), ex.what());

194bool BaseControllerInterface::get_predicate(const std::string& predicate_name) const {
195 auto predicate_it = predicates_.find(predicate_name);
196 if (predicate_it == predicates_.end()) {
197 RCLCPP_ERROR_THROTTLE(
198 get_node()->get_logger(), *get_node()->get_clock(), 1000,
199 "Failed to get predicate '%s': Predicate does not exists, returning false.", predicate_name.c_str());
200 return false;
201 }
202 try {
203 return predicate_it->second.get_value();
204 } catch (const std::exception& ex) {
205 RCLCPP_ERROR_THROTTLE(
206 get_node()->get_logger(), *get_node()->get_clock(), 1000,
207 "Failed to evaluate callback of predicate '%s', returning false: %s", predicate_name.c_str(), ex.what());
208 }
209 return false;

277void BaseControllerInterface::publish_predicates() {
278 auto message(predicate_message_);
279 for (auto predicate_it = predicates_.begin(); predicate_it != predicates_.end(); ++predicate_it) {
280 if (auto new_predicate = predicate_it->second.query(); new_predicate) {
281 message.predicates.push_back(get_predicate_message(predicate_it->first, *new_predicate));
282 }
283 }
284 if (message.predicates.size()) {
285 predicate_publisher_->publish(message);
286 }

289std::string BaseControllerInterface::validate_and_declare_signal(
290 const std::string& signal_name, const std::string& type, const std::string& default_topic, bool fixed_topic) {
291 auto parsed_signal_name = modulo_utils::parsing::parse_topic_name(signal_name);
292 if (parsed_signal_name.empty()) {
293 RCLCPP_WARN(
294 get_node()->get_logger(),
295 "The parsed signal name for %s '%s' is empty. Provide a string with valid characters for the signal name "
296 "([a-zA-Z0-9_]).",
297 type.c_str(), signal_name.c_str());
298 return "";
299 }
300 if (signal_name != parsed_signal_name) {
301 RCLCPP_WARN(
302 get_node()->get_logger(),
303 "The parsed signal name for %s '%s' is '%s'. Use the parsed signal name to refer to this %s and its topic "
304 "parameter.",
305 type.c_str(), signal_name.c_str(), parsed_signal_name.c_str(), type.c_str());
306 }
307 if (inputs_.find(parsed_signal_name) != inputs_.end()) {
308 RCLCPP_WARN(get_node()->get_logger(), "Signal '%s' already exists as input.", parsed_signal_name.c_str());
309 return "";
310 }
311 if (outputs_.find(parsed_signal_name) != outputs_.end()) {
312 RCLCPP_WARN(get_node()->get_logger(), "Signal '%s' already exists as output", parsed_signal_name.c_str());
313 return "";
314 }
315 auto topic = default_topic.empty() ? "~/" + parsed_signal_name : default_topic;
316 auto parameter_name = parsed_signal_name + "_topic";
317 if (get_node()->has_parameter(parameter_name) && get_parameter(parameter_name)->is_empty()) {
318 set_parameter_value<std::string>(parameter_name, topic);
319 } else {
320 add_parameter<std::string>(
321 parameter_name, topic, "Signal topic name of " + type + " '" + parsed_signal_name + "'", fixed_topic);
322 }
323 RCLCPP_DEBUG(
324 get_node()->get_logger(), "Declared %s '%s' and parameter '%s' with value '%s'.", type.c_str(),
325 parsed_signal_name.c_str(), parameter_name.c_str(), topic.c_str());
326 return parsed_signal_name;
327}

468void BaseControllerInterface::add_service(
469 const std::string& service_name, const std::function<ControllerServiceResponse(void)>& callback) {
470 auto parsed_service_name = validate_service_name(service_name, "empty");
471 if (!parsed_service_name.empty()) {
472 try {
473 auto service = get_node()->create_service<modulo_interfaces::srv::EmptyTrigger>(
474 "~/" + parsed_service_name,
475 [this, callback](
476 const std::shared_ptr<modulo_interfaces::srv::EmptyTrigger::Request>,
477 std::shared_ptr<modulo_interfaces::srv::EmptyTrigger::Response> response) {
478 try {
479 if (this->command_mutex_.try_lock_for(100ms)) {
480 auto callback_response = callback();
481 this->command_mutex_.unlock();
482 response->success = callback_response.success;
483 response->message = callback_response.message;
484 } else {
485 response->success = false;
486 response->message = "Unable to acquire lock for command interface within 100ms";
487 }
488 } catch (const std::exception& ex) {
489 response->success = false;
490 response->message = ex.what();
491 }
492 },
493 qos_);
494 empty_services_.insert_or_assign(parsed_service_name, service);
495 } catch (const std::exception& ex) {
496 RCLCPP_ERROR(get_node()->get_logger(), "Failed to add service '%s': %s", parsed_service_name.c_str(), ex.what());
497 }
498 }
