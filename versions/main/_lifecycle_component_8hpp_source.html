281inline void LifecycleComponent::add_output(
282 const std::string& signal_name, const std::shared_ptr<DataT>& data, const std::string& default_topic,
283 bool fixed_topic, bool publish_on_step
284) {
285 if (this->get_lifecycle_state().id() != lifecycle_msgs::msg::State::PRIMARY_STATE_UNCONFIGURED
286 && this->get_lifecycle_state().id() != lifecycle_msgs::msg::State::PRIMARY_STATE_INACTIVE) {
287 RCLCPP_WARN_STREAM_THROTTLE(this->get_logger(), *this->get_clock(), 1000,
288 "Adding output in state " << this->get_lifecycle_state().label() << " is not allowed.");
289 return;
290 }
291 try {
292 this->create_output(
293 modulo_core::communication::PublisherType::LIFECYCLE_PUBLISHER, signal_name, data, default_topic, fixed_topic,
294 publish_on_step);
295 } catch (const modulo_core::exceptions::AddSignalException& ex) {
296 RCLCPP_ERROR_STREAM(this->get_logger(), "Failed to add output '" << signal_name << "': " << ex.what());
297 }
298}
