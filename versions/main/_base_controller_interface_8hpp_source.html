602template<typename T>
603inline std::optional<T> BaseControllerInterface::read_input(const std::string& name) {
604 if (!check_input_valid(name)) {
605 return {};
606 }
607 auto message =
608 **std::get<realtime_tools::RealtimeBuffer<std::shared_ptr<modulo_core::EncodedState>>>(inputs_.at(name).buffer)
609 .readFromNonRT();
610 std::shared_ptr<state_representation::State> state;
611 try {
612 auto message_pair = input_message_pairs_.at(name);
613 message_pair->read<modulo_core::EncodedState, state_representation::State>(message);
614 state = message_pair->get_message_pair<modulo_core::EncodedState, state_representation::State>()->get_data();
615 } catch (const std::exception& ex) {
616 RCLCPP_WARN_THROTTLE(
617 get_node()->get_logger(), *get_node()->get_clock(), 1000,
618 "Could not read EncodedState message on input '%s': %s", name.c_str(), ex.what());
619 return {};
620 }
621 if (state->is_empty()) {
622 return {};
623 }
624 auto cast_ptr = std::dynamic_pointer_cast<T>(state);
625 if (cast_ptr != nullptr) {
626 return *cast_ptr;
627 } else {
628 RCLCPP_WARN_THROTTLE(
629 get_node()->get_logger(), *get_node()->get_clock(), 1000,
630 "Dynamic cast of message on input '%s' from type '%s' to type '%s' failed.", name.c_str(),
631 get_state_type_name(state->get_type()).c_str(), get_state_type_name(T().get_type()).c_str());
632 }
633 return {};
634}

690template<typename T>
691inline void BaseControllerInterface::write_output(const std::string& name, const T& data) {
692 if (data.is_empty()) {
693 RCLCPP_DEBUG_THROTTLE(
694 get_node()->get_logger(), *get_node()->get_clock(), 1000,
695 "Skipping publication of output '%s' due to emptiness of state", name.c_str());
696 return;
697 }
698 if (outputs_.find(name) == outputs_.end()) {
699 RCLCPP_WARN_THROTTLE(
700 get_node()->get_logger(), *get_node()->get_clock(), 1000, "Could not find output '%s'", name.c_str());
701 return;
702 }
703 EncodedStatePublishers publishers;
704 try {
705 publishers = std::get<EncodedStatePublishers>(outputs_.at(name));
706 } catch (const std::bad_variant_access&) {
707 RCLCPP_WARN_THROTTLE(
708 get_node()->get_logger(), *get_node()->get_clock(), 1000,
709 "Could not retrieve publisher for output '%s': Invalid output type", name.c_str());
710 return;
711 }
712 if (const auto output_type = std::get<0>(publishers)->get_type(); output_type != data.get_type()) {
713 RCLCPP_WARN_THROTTLE(
714 get_node()->get_logger(), *get_node()->get_clock(), 1000,
715 "Skipping publication of output '%s' due to wrong data type (expected '%s', got '%s')",
716 state_representation::get_state_type_name(output_type).c_str(),
717 state_representation::get_state_type_name(data.get_type()).c_str(), name.c_str());
718 return;
719 }
720 auto rt_pub = std::get<2>(publishers);
721 if (rt_pub && rt_pub->trylock()) {
722 try {
723 modulo_core::translators::write_message<T>(rt_pub->msg_, data, get_node()->get_clock()->now());
724 } catch (const modulo_core::exceptions::MessageTranslationException& ex) {
725 RCLCPP_ERROR_THROTTLE(
726 get_node()->get_logger(), *get_node()->get_clock(), 1000, "Failed to publish output '%s': %s", name.c_str(),
727 ex.what());
728 }
729 rt_pub->unlockAndPublish();
730 }
731}

733template<typename PublisherT, typename MsgT, typename T>
734void BaseControllerInterface::write_std_output(const std::string& name, const T& data) {
735 if (outputs_.find(name) == outputs_.end()) {
736 RCLCPP_WARN_THROTTLE(
737 get_node()->get_logger(), *get_node()->get_clock(), 1000, "Could not find output '%s'", name.c_str());
738 return;
739 }
740 PublisherT publishers;
741 try {
742 publishers = std::get<PublisherT>(outputs_.at(name));
743 } catch (const std::bad_variant_access&) {
744 RCLCPP_WARN_THROTTLE(
745 get_node()->get_logger(), *get_node()->get_clock(), 1000,
746 "Could not retrieve publisher for output '%s': Invalid output type", name.c_str());
747 return;
748 }
749 auto rt_pub = publishers.second;
750 if (rt_pub && rt_pub->trylock()) {
751 rt_pub->msg_.data = data;
752 rt_pub->unlockAndPublish();
753 }
754}
