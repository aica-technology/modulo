658template<typename T>
659inline void BaseControllerInterface::add_input(const std::string& name, const std::string& topic_name) {
660 if constexpr (modulo_core::concepts::CustomT<T>) {
661 auto buffer = std::make_shared<realtime_tools::RealtimeBuffer<std::shared_ptr<T>>>();
662 auto input = ControllerInput(buffer);
663 auto parsed_name = validate_and_declare_signal(name, "input", topic_name);
664 if (!parsed_name.empty()) {
665 inputs_.insert_or_assign(parsed_name, input);
666 custom_input_configuration_callables_.insert_or_assign(
667 name, [this](const std::string& name, const std::string& topic) {
668 auto subscription =
669 get_node()->create_subscription<T>(topic, qos_, [this, name](const std::shared_ptr<T> message) {
670 auto buffer_variant = std::get<std::any>(inputs_.at(name).buffer);
671 auto buffer = std::any_cast<std::shared_ptr<realtime_tools::RealtimeBuffer<std::shared_ptr<T>>>>(
672 buffer_variant);
673 buffer->writeFromNonRT(message);
674 inputs_.at(name).timestamp = std::chrono::steady_clock::now();
675 });
676 subscriptions_.push_back(subscription);
677 });
678 }
679 } else {
680 auto buffer = realtime_tools::RealtimeBuffer<std::shared_ptr<modulo_core::EncodedState>>();
681 auto input = ControllerInput(buffer);
682 auto parsed_name = validate_and_declare_signal(name, "input", topic_name);
683 if (!parsed_name.empty()) {
684 inputs_.insert_or_assign(parsed_name, input);
685 input_message_pairs_.insert_or_assign(
686 parsed_name,
687 modulo_core::communication::make_shared_message_pair(std::make_shared<T>(), get_node()->get_clock()));
688 }
689 }
690}

783template<typename T>
784inline std::optional<T> BaseControllerInterface::read_input(const std::string& name) {
785 if (!check_input_valid(name)) {
786 return {};
787 }
788
789 if constexpr (modulo_core::concepts::CustomT<T>) {
790 try {
791 auto buffer_variant = std::get<std::any>(inputs_.at(name).buffer);
792 auto buffer = std::any_cast<std::shared_ptr<realtime_tools::RealtimeBuffer<std::shared_ptr<T>>>>(buffer_variant);
793 return **(buffer->readFromNonRT());
794 } catch (const std::bad_any_cast& ex) {
795 RCLCPP_ERROR(get_node()->get_logger(), "Failed to read custom input: %s", ex.what());
796 }
797 return {};
798 } else {
799 auto message =
800 **std::get<realtime_tools::RealtimeBuffer<std::shared_ptr<modulo_core::EncodedState>>>(inputs_.at(name).buffer)
801 .readFromNonRT();
802 std::shared_ptr<state_representation::State> state;
803 try {
804 auto message_pair = input_message_pairs_.at(name);
805 message_pair->read<modulo_core::EncodedState, state_representation::State>(message);
806 state = message_pair->get_message_pair<modulo_core::EncodedState, state_representation::State>()->get_data();
807 } catch (const std::exception& ex) {
808 RCLCPP_WARN_THROTTLE(
809 get_node()->get_logger(), *get_node()->get_clock(), 1000,
810 "Could not read EncodedState message on input '%s': %s", name.c_str(), ex.what());
811 return {};
812 }
813 if (state->is_empty()) {
814 return {};
815 }
816 auto cast_ptr = std::dynamic_pointer_cast<T>(state);
817 if (cast_ptr != nullptr) {
818 return *cast_ptr;
819 } else {
820 RCLCPP_WARN_THROTTLE(
821 get_node()->get_logger(), *get_node()->get_clock(), 1000,
822 "Dynamic cast of message on input '%s' from type '%s' to type '%s' failed.", name.c_str(),
823 get_state_type_name(state->get_type()).c_str(), get_state_type_name(T().get_type()).c_str());
824 }
825 return {};
826 }
827}

883template<typename T>
884inline void BaseControllerInterface::write_output(const std::string& name, const T& data) {
885 if (outputs_.find(name) == outputs_.end()) {
886 RCLCPP_WARN_THROTTLE(
887 get_node()->get_logger(), *get_node()->get_clock(), 1000, "Could not find output '%s'", name.c_str());
888 return;
889 }
890
891 if constexpr (modulo_core::concepts::CustomT<T>) {
892 CustomPublishers publishers;
893 try {
894 publishers = std::get<CustomPublishers>(outputs_.at(name));
895 } catch (const std::bad_variant_access&) {
896 RCLCPP_WARN_THROTTLE(
897 get_node()->get_logger(), *get_node()->get_clock(), 1000,
898 "Could not retrieve publisher for output '%s': Invalid output type", name.c_str());
899 return;
900 }
901
902 std::shared_ptr<realtime_tools::RealtimePublisher<T>> rt_pub;
903 try {
904 rt_pub = std::any_cast<std::shared_ptr<realtime_tools::RealtimePublisher<T>>>(publishers.second);
905 } catch (const std::bad_any_cast& ex) {
906 RCLCPP_ERROR_THROTTLE(
907 get_node()->get_logger(), *get_node()->get_clock(), 1000,
908 "Skipping publication of output '%s' due to wrong data type: %s", name.c_str(), ex.what());
909 return;
910 }
911 if (rt_pub && rt_pub->trylock()) {
912 rt_pub->msg_ = data;
913 rt_pub->unlockAndPublish();
914 }
915 } else {
916 if (data.is_empty()) {
917 RCLCPP_DEBUG_THROTTLE(
918 get_node()->get_logger(), *get_node()->get_clock(), 1000,
919 "Skipping publication of output '%s' due to emptiness of state", name.c_str());
920 return;
921 }
922 EncodedStatePublishers publishers;
923 try {
924 publishers = std::get<EncodedStatePublishers>(outputs_.at(name));
925 } catch (const std::bad_variant_access&) {
926 RCLCPP_WARN_THROTTLE(
927 get_node()->get_logger(), *get_node()->get_clock(), 1000,
928 "Could not retrieve publisher for output '%s': Invalid output type", name.c_str());
929 return;
930 }
931 if (const auto output_type = std::get<0>(publishers)->get_type(); output_type != data.get_type()) {
932 RCLCPP_WARN_THROTTLE(
933 get_node()->get_logger(), *get_node()->get_clock(), 1000,
934 "Skipping publication of output '%s' due to wrong data type (expected '%s', got '%s')",
935 state_representation::get_state_type_name(output_type).c_str(),
936 state_representation::get_state_type_name(data.get_type()).c_str(), name.c_str());
937 return;
938 }
939 auto rt_pub = std::get<2>(publishers);
940 if (rt_pub && rt_pub->trylock()) {
941 try {
942 modulo_core::translators::write_message<T>(rt_pub->msg_, data, get_node()->get_clock()->now());
943 } catch (const modulo_core::exceptions::MessageTranslationException& ex) {
944 RCLCPP_ERROR_THROTTLE(
945 get_node()->get_logger(), *get_node()->get_clock(), 1000, "Failed to publish output '%s': %s", name.c_str(),
946 ex.what());
947 }
948 rt_pub->unlockAndPublish();
949 }
950 }
951}

1001template<bool acquire_lock, typename CallbackT>
1002inline void
1003BaseControllerInterface::create_service(const std::string& service_name, const std::function<CallbackT>& callback)
1004 requires(
1005 std::is_same_v<CallbackT, ControllerServiceResponse(const std::string&)>
1006 || std::is_same_v<CallbackT, ControllerServiceResponse(void)>)
1007{
1008 constexpr bool is_empty = std::is_same_v<CallbackT, ControllerServiceResponse(void)>;
1009 using ServiceT =
1010 std::conditional_t<is_empty, modulo_interfaces::srv::EmptyTrigger, modulo_interfaces::srv::StringTrigger>;
1011
1012 auto parsed_service_name = validate_service_name(service_name, is_empty ? "empty" : "string");
1013 if (parsed_service_name.empty()) {
1014 return;
1015 }
1016
1017 try {
1018 auto service = get_node()->create_service<ServiceT>(
1019 "~/" + parsed_service_name,
1020 [this, callback](
1021 const std::shared_ptr<typename ServiceT::Request> request,
1022 std::shared_ptr<typename ServiceT::Response> response) {
1023 try {
1024 auto run_callback = [&]() {
1025 ControllerServiceResponse callback_response;
1026 if constexpr (is_empty) {
1027 callback_response = callback();
1028 } else {
1029 callback_response = callback(request->payload);
1030 }
1031 response->success = callback_response.success;
1032 response->message = callback_response.message;
1033 };
1034
1035 if constexpr (!acquire_lock) {
1036 run_callback();
1037 } else {
1038 std::unique_lock<std::timed_mutex> lock(this->command_mutex_, std::defer_lock);
1039 if (lock.try_lock_for(std::chrono::milliseconds{100})) {
1040 run_callback();
1041 } else {
1042 response->success = false;
1043 response->message = "Unable to acquire lock for command interface within 100ms";
1044 }
1045 }
1046 } catch (const std::exception& ex) {
1047 response->success = false;
1048 response->message = ex.what();
1049 }
1050 },
1051 qos_);
1052 if constexpr (is_empty) {
1053 empty_services_.insert_or_assign(parsed_service_name, service);
1054 } else {
1055 string_services_.insert_or_assign(parsed_service_name, service);
1056 }
1057 } catch (const std::exception& ex) {
1058 RCLCPP_ERROR(get_node()->get_logger(), "Failed to add service '%s': %s", parsed_service_name.c_str(), ex.what());
1059 }
1060}

1062template<typename T>
1063inline void BaseControllerInterface::publish_transforms(
1064 const std::vector<state_representation::CartesianPose>& transforms, const std::shared_ptr<T>& tf_broadcaster,
1065 bool is_static) {
1066 std::string modifier = is_static ? "static " : "";
1067 if (tf_broadcaster == nullptr) {
1068 RCLCPP_ERROR_STREAM_THROTTLE(
1069 this->get_node()->get_logger(), *this->get_node()->get_clock(), 1000,
1070 "Failed to send " << modifier << "transform: No " << modifier << "TF broadcaster configured.");
1071 return;
1072 }
1073 try {
1074 std::vector<geometry_msgs::msg::TransformStamped> transform_messages;
1075 transform_messages.reserve(transforms.size());
1076 for (const auto& tf : transforms) {
1077 geometry_msgs::msg::TransformStamped transform_message;
1078 modulo_core::translators::write_message(transform_message, tf, this->get_node()->get_clock()->now());
1079 transform_messages.emplace_back(transform_message);
1080 }
1081 tf_broadcaster->sendTransform(transform_messages);
1082 } catch (const std::exception& ex) {
1083 RCLCPP_ERROR_STREAM_THROTTLE(
1084 this->get_node()->get_logger(), *this->get_node()->get_clock(), 1000,
1085 "Failed to send " << modifier << "transform: " << ex.what());
1086 }
