269template<typename DataT>
270inline void LifecycleComponent::add_output(
271 const std::string& signal_name, const std::shared_ptr<DataT>& data, const std::string& default_topic,
272 bool fixed_topic, bool publish_on_step
273) {
274 if (this->get_current_state().id() != lifecycle_msgs::msg::State::PRIMARY_STATE_UNCONFIGURED
275 && this->get_current_state().id() != lifecycle_msgs::msg::State::PRIMARY_STATE_INACTIVE) {
276 RCLCPP_WARN_STREAM_THROTTLE(
277 this->get_logger(), *this->get_clock(), 1000,
278 "Adding output in state " << this->get_current_state().label() << " is not allowed.");
279 return;
280 }
281 try {
282 this->create_output(signal_name, data, default_topic, fixed_topic, publish_on_step);
283 } catch (const exceptions::AddSignalException& ex) {
284 RCLCPP_ERROR_STREAM(this->get_logger(), "Failed to add output '" << signal_name << "': " << ex.what());
285 }
286}
